❖ SSH — OpenSSH Server
Remote shell access, key management, port forwarding

SSH — OpenSSH Server

Config file: /etc/ssh/sshd_config

# Connect from a modern system (SSH key algorithms required for old OpenSSH 5.x):
ssh -o HostKeyAlgorithms=+ssh-rsa -o PubkeyAcceptedAlgorithms=+ssh-rsa sysadmin@192.168.1.234

# Add to your ~/.ssh/config for convenience:
Host etrayz
    HostName 192.168.1.234
    User sysadmin
    HostKeyAlgorithms +ssh-rsa
    PubkeyAcceptedAlgorithms +ssh-rsa

# Change SSH port:
sudo sed -i 's/^Port .*/Port 2222/' /etc/ssh/sshd_config
sudo /etc/init.d/ssh reload

# Disable root login:
sudo sed -i 's/^PermitRootLogin .*/PermitRootLogin no/' /etc/ssh/sshd_config
sudo /etc/init.d/ssh reload

# Add SSH key (from your PC):
ssh-copy-id -o HostKeyAlgorithms=+ssh-rsa sysadmin@192.168.1.234
# Or manually:
cat ~/.ssh/id_rsa.pub | ssh sysadmin@etrayz 'cat >> ~/.ssh/authorized_keys'

# Disable password auth (key-only):
sudo sed -i 's/^PasswordAuthentication .*/PasswordAuthentication no/' /etc/ssh/sshd_config
sudo /etc/init.d/ssh reload

# SCP file transfer:
scp -o HostKeyAlgorithms=+ssh-rsa file.txt sysadmin@192.168.1.234:/home/public/

# Port forwarding (access NAS web from remote):
ssh -L 8070:localhost:70 sysadmin@etrayz
# Then open http://localhost:8070 in browser
⇄ Samba — Network File Sharing
SMBv1 shares, user management, access control

Samba — Network File Sharing (SMBv1)

Config file: /etc/samba/smb.conf

Note: Samba 3.x only supports SMBv1. Modern clients need configuration.

# Test connection from Linux:
smbclient -L //192.168.1.234 --option='client min protocol=NT1' -N

# Mount a share on Linux:
sudo mount -t cifs //192.168.1.234/public /mnt/nas \
  -o vers=1.0,guest,uid=$(id -u),gid=$(id -g)

# Permanent mount (add to /etc/fstab):
//192.168.1.234/home /mnt/nas-home cifs vers=1.0,user=sysadmin,pass=etrayz,uid=1000 0 0

# Linux client config (~/.smb/smb.conf or /etc/samba/smb.conf):
[global]
   client min protocol = NT1

# Windows 10/11 — enable SMBv1:
# Control Panel → Programs → Turn Windows features on/off
# → SMB 1.0/CIFS File Sharing Support → SMB 1.0 Client

# Add a new share on NAS:
sudo nano /etc/samba/smb.conf
# Add:
[media]
   path = /home/public/Videos
   browseable = yes
   writable = yes
   guest ok = yes
   force user = sysadmin

# Reload Samba:
sudo /etc/init.d/samba reload

# Change sysadmin Samba password:
sudo smbpasswd -a sysadmin

# Check connected clients:
smbstatus

# List shares:
smbclient -L localhost -N
⇩ Transmission — BitTorrent Client
Download management, speed limits, remote access

Transmission — BitTorrent Client

Config file: /etc/transmission-daemon/settings.json (edit while stopped!)

Web UI: http://192.168.1.234:9091

# IMPORTANT: Stop transmission before editing config!
sudo /etc/init.d/transmission-daemon stop

# Edit settings:
sudo nano /etc/transmission-daemon/settings.json

# Key settings:
#   "download-dir": "/home/downloads/complete"
#   "incomplete-dir": "/home/downloads/incomplete"
#   "speed-limit-up": 500          (KB/s)
#   "speed-limit-up-enabled": true
#   "speed-limit-down": 0          (0 = unlimited)
#   "ratio-limit": 2.0
#   "peer-port": 51413
#   "rpc-whitelist": "127.0.0.1,192.168.*.*"

# Start after editing:
sudo /etc/init.d/transmission-daemon start

# Add torrent from CLI:
transmission-remote -a "magnet:?xt=urn:..."
transmission-remote -a /path/to/file.torrent

# List active torrents:
transmission-remote -l

# Pause all torrents:
transmission-remote -t all -S

# Resume all:
transmission-remote -t all -s

# Remove completed (keep files):
transmission-remote -t 1 -r

# Watch directory (auto-add .torrent files):
# Place .torrent files in /home/downloads/watch/

# Logs:
tail -f /var/log/transmission-daemon/transmission-daemon.log
⤓ aria2 — Download Manager
HTTP/FTP/BitTorrent/Magnet downloads, XML-RPC API

aria2 — Multi-Protocol Download Manager

Config file: /etc/aria2/aria2.conf

Session file: /etc/aria2/aria2.session (active downloads persist across restarts)

Web UI: http://192.168.1.234:70/cgi-bin/aria2-ui.sh

Protocols: HTTP, HTTPS, FTP, BitTorrent, Magnet links, Metalink

Basic Downloads

# Download a file:
aria2c "http://example.com/file.zip"

# Download with multiple connections (faster):
aria2c -x 4 -s 4 "http://example.com/largefile.iso"

# Download magnet link:
aria2c "magnet:?xt=urn:btih:..."

# Download from text file (one URL per line):
aria2c -i urls.txt

Premium File Hosting (Uploaded.net, 1fichier, etc.)

aria2 supports HTTPS, HTTP authentication, cookies, and custom headers — everything needed to download from premium file hosting services. Since these sites use browser-based login, the easiest method is to export your login cookies:

# Step-by-step: Download from premium file hosters
#
# 1. Install a cookie export extension in your browser:
#    - Firefox: "Cookie Quick Manager" or "cookies.txt"
#    - Chrome:  "Get cookies.txt LOCALLY"
#
# 2. Log in to the file hosting site (e.g. uploaded.net,
#    1fichier.com, rapidgator.net, etc.)
#
# 3. Export cookies to a text file (Netscape/wget format).
#    The file looks like this:
#    .uploaded.net  TRUE  /  TRUE  0  login  myuser
#    .uploaded.net  TRUE  /  TRUE  0  auth   a1b2c3d4e5
#
# 4. In the aria2 Web UI, click "Authentication & options":
#    - Paste the cookie file content into the Cookies field
#    - Paste the download link into the URL field
#    - Click "+ Add"
#
# Or from the command line:
aria2c --load-cookies=cookies.txt "https://uploaded.net/file/abc123"

HTTP / FTP Authentication

# For sites that use HTTP Basic/Digest auth:
aria2c --http-user=mylogin --http-passwd=mypass \
  "https://example.com/protected/file.zip"

# FTP login:
aria2c --ftp-user=mylogin --ftp-passwd=mypass \
  "ftp://ftp.example.com/pub/file.tar.gz"

# Custom referer (some hosters check this):
aria2c --referer="https://filehost.com/download-page" \
  "https://filehost.com/dl/abc123/file.zip"

# Custom headers:
aria2c --header="Authorization: Bearer token123" \
  "https://api.example.com/files/download"

# All options combined:
aria2c --http-user=me --http-passwd=secret \
  --referer="https://host.com/page" \
  --load-cookies=cookies.txt \
  -x 4 -s 4 -o "output-name.zip" \
  "https://host.com/download/abc123"

XML-RPC API (remote control)

# Add download via XML-RPC (from another machine):
curl -H 'Content-Type: text/xml' -d '<?xml version="1.0"?>
<methodCall><methodName>aria2.addUri</methodName>
<params><param><value><string>http://example.com/file.zip</string></value></param>
</params></methodCall>' http://192.168.1.234:6800/rpc

# Clear completed downloads:
curl -H 'Content-Type: text/xml' -d '<?xml version="1.0"?>
<methodCall><methodName>aria2.purgeDownloadResult</methodName>
<params></params></methodCall>' http://localhost:6800/rpc

Configuration & Service

# Config options (/etc/aria2/aria2.conf):
#   dir=/home/downloads/complete      Download directory
#   max-concurrent-downloads=3        Parallel downloads
#   max-connection-per-server=4       Connections per file
#   split=4                           Split file into N parts
#   max-overall-upload-limit=100K     Upload speed cap
#   continue=true                     Resume partial downloads
#   seed-ratio=0                      Don't seed torrents (0=disable)

# Restart service:
sudo /etc/init.d/aria2 restart

# Check logs:
tail -f /var/log/aria2.log

MiniDLNA — UPnP/DLNA Media Server

Config file: /etc/minidlna.conf

Database: /var/lib/minidlna/files.db

Status page: http://192.168.1.234:8200

# Media directories (configured in /etc/minidlna.conf):
#   media_dir=A,/home/public/Music       Audio files
#   media_dir=V,/home/public/Videos      Video files
#   media_dir=P,/home/public/Pictures    Image files
#   media_dir=/home/public               All media types

# Add media files:
# Via Samba: Copy to \\etrayz\public\Music, Videos, or Pictures
# Via SCP:
scp movie.mp4 sysadmin@192.168.1.234:/home/public/Videos/
scp -r "Music Album/" sysadmin@192.168.1.234:/home/public/Music/

# Force rescan (after adding many files):
sudo /etc/init.d/minidlna force-reload

# Full database rebuild:
sudo /etc/init.d/minidlna stop
sudo rm /var/lib/minidlna/files.db
sudo /etc/init.d/minidlna start

# Check status:
curl http://localhost:8200

# Supported formats:
# Video: MP4, MKV, AVI, MPEG, WMV, FLV, MOV
# Audio: MP3, FLAC, OGG, WAV, WMA, AAC
# Images: JPEG, PNG, GIF, BMP, TIFF

# Clients that can play from DLNA:
# - VLC (View → Playlist → Local Network → UPnP)
# - Samsung/LG/Sony Smart TVs (Media/AllShare)
# - Kodi/XBMC
# - Windows Media Player (Stream → More streaming options)
# - BubbleUPnP (Android)

# Inotify (auto-detect new files):
# Set inotify=yes in config (default). New files appear within ~30s.
# If inotify stops working on large libraries, increase limit:
echo 65536 | sudo tee /proc/sys/fs/inotify/max_user_watches

# Logs:
tail -f /var/log/minidlna.log

Config: /etc/lighttpd/lighttpd.conf

Dashboard: port 70 — /var/www/

Personal site: port 80 — /var/www-site/

# Dual-port config (/etc/lighttpd/conf-enabled/20-etrayz.conf):
$SERVER["socket"] == ":80" {
    server.document-root = "/var/www-site"

# Edit your personal website:
sudo nano /var/www-site/index.html
# Or via Samba: \\etrayz\home → navigate to /var/www-site/

# Add a new CGI script:
sudo nano /var/www/cgi-bin/myscript.sh
sudo chmod 755 /var/www/cgi-bin/myscript.sh
# Access at: http://etrayz:70/cgi-bin/myscript.sh

# CGI script template:
#!/bin/sh
echo "Content-Type: text/html"

♥ monit — Process Monitor

monit — Process Monitor & Watchdog

Config file: /etc/monit/monitrc (chmod 600)

Web UI: http://192.168.1.234:70/cgi-bin/monit-proxy.sh

# Check all services:
sudo monit summary

# Start/stop/restart a service via monit:
sudo monit start transmission
sudo monit stop minidlna
sudo monit restart sshd

# Reload monit config:
sudo monit reload

# View monit log:
tail -f /var/log/monit.log

# Add a new process to monitor:
sudo nano /etc/monit/monitrc
# Add:
check process myapp with pidfile /var/run/myapp.pid
    start program = "/etc/init.d/myapp start"
    stop program  = "/etc/init.d/myapp stop"
    if failed port 1234 then restart
    if 3 restarts within 5 cycles then timeout

# Monitor a file:
check file mylog with path /var/log/myapp.log
    if size > 100 MB then exec "/usr/bin/logrotate /etc/logrotate.d/myapp"

# Monitor disk space:
check filesystem data with path /dev/sda3
    if space usage > 90% then alert

# System resource alerts:
check system etrayz
    if loadavg (5min) > 4 then alert
    if memory usage > 90% then alert

# Monit checks every 60 seconds (configurable):
set daemon 60

# Currently monitored:
# sshd, smbd, nmbd, transmission, lighttpd, minidlna, aria2, usbcopy
# Filesystems: rootfs (/dev/md0), datafs (/dev/sda3)
# System: load average, memory usage
⚙ System Administration
Package management, networking, disk tools, cron jobs

System Administration

# === Package Management ===
sudo apt-get update                   # Refresh package list
sudo apt-get install package-name     # Install a package
sudo apt-get remove package-name      # Remove a package
dpkg -l                               # List installed packages
dpkg -L package-name                  # List files in a package

# === Disk Management ===
df -h                                 # Disk usage
du -sh /home/*                        # Directory sizes
sudo smartctl -a /dev/sda             # SMART disk health
sudo xfs_repair /dev/sda3             # Repair XFS (unmount first!)
sudo hdparm -I /dev/sda               # Disk info

# === Network ===
ifconfig eth0                         # IP address
route -n                              # Routing table
cat /etc/resolv.conf                  # DNS servers
ping -c 3 google.com                  # Test connectivity
netstat -tlnp                         # Listening ports

# === Process Management ===
htop                                  # Interactive process viewer
ps aux                                # List all processes
kill PID                              # Stop a process
sudo /etc/init.d/SERVICE restart      # Restart a service

# === Cron Jobs ===
crontab -e                            # Edit user cron
sudo crontab -e                       # Edit root cron
# Example: backup every night at 2am:
0 2 * * * rsync -a /home/public/ /home2/backup/

# === Logs ===
tail -f /var/log/messages             # System log
tail -f /var/log/auth.log             # Login attempts
tail -f /var/log/daemon.log           # Service logs
dmesg                                 # Kernel messages

# === Settings System ===
# All settings stored in: /etc/etrayz/settings.conf
# Factory defaults:       /etc/etrayz/defaults.conf
# Apply all settings:     sudo etrayz-apply all
# Apply one section:      sudo etrayz-apply samba

# === GPIO (LEDs, Buzzer, Buttons) ===
# LEDs:
echo 1 > /sys/gpio/devices/led_sys       # System LED on
echo 0 > /sys/gpio/devices/led_sata1     # SATA1 LED off
# Buzzer:
etrayz-buzzer single                      # Beep patterns: single, double,
etrayz-buzzer r2d2                        # triple, long, sos, startup, r2d2, imperial

# === Backup ===
# Backup to second HDD:
rsync -av --delete /home/ /home2/backup-home/
# Backup to external USB drive:
rsync -av /home/public/ /media/usb/backup/

# === User Management ===
passwd sysadmin                       # Change password
sudo adduser newuser                  # Add new user
sudo smbpasswd -a newuser             # Add Samba user
